/**

 * - Ícone em círculo colorido
 * - Título e descrição
 * - Badge de destaque
 * - Lista de features/tecnologias
 *

 *

 * @param {Service} props.service - Dados do serviço

 *
 * @example
 * <ServiceCard service={{
 *   title: "Web Dev",
 *   description: "Sites modernos",
 *   icon: Code,
 *   badge: "Popular",
 *   features: ["React", "Next.js"]
 * }} />
